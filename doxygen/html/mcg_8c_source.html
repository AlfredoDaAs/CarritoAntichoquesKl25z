18 /* Functon name : pll_init
19  *
20  * Mode transition: Option to move from FEI to PEE mode or to just initialize the PLL
21  *
22  * This function initializess PLL0. Either OSC0 is selected for the
23  * reference clock source. The oscillators can be configured to use a crystal or take in an
24  * external square wave clock.
25  * Using the function parameter names the PLL frequency is calculated as follows:
26  * PLL freq = ((crystal_val / prdiv_val) * vdiv_val)
27  * Refer to the readme file in the mcg driver directory for examples of pll_init configurations.
28  * All parameters must be provided, for example crystal_val must be provided even if the
29  * oscillator associated with that parameter is already initialized.
30  * The various passed parameters are checked to ensure they are within the allowed range. If any
31  * of these checks fail the driver will exit and return a fail/error code. An error code will
32  * also be returned if any error occurs during the PLL initialization sequence. Refer to the
33  * readme file in the mcg driver directory for a list of all these codes.
34  *
35  * Parameters: crystal_val - external clock frequency in Hz either from a crystal or square
36  * wave clock source
37  * hgo_val - selects whether low power or high gain mode is selected
38  * for the crystal oscillator. This has no meaning if an
39  * external clock is used.
40  * erefs_val - selects external clock (=0) or crystal osc (=1)
41  * prdiv_val - value to divide the external clock source by to create the desired
42  * PLL reference clock frequency
43  * vdiv_val - value to multiply the PLL reference clock frequency by
44  * mcgout_select - 0 if the PLL is just to be enabled, non-zero if the PLL is used
45  * to provide the MCGOUT clock for the system.
46  *
47  * Return value : PLL frequency (Hz) or error code
48  */
49 
50 int pll_init(int crystal_val, unsigned char hgo_val, unsigned char erefs_val, signed char prdiv_val, signed char vdiv_val, unsigned char mcgout_select)
51 {
52  unsigned char frdiv_val;
53  unsigned char temp_reg;
54  unsigned char prdiv, vdiv;
55  short i;
56  int ref_freq;
57  int pll_freq;
58 
59  // check if in FEI mode
60  if (!((((MCG_S & MCG_S_CLKST_MASK) >> MCG_S_CLKST_SHIFT) == 0x0) && // check CLKS mux has selcted FLL output
61  (MCG_S & MCG_S_IREFST_MASK) && // check FLL ref is internal ref clk
62  (!(MCG_S & MCG_S_PLLST_MASK)))) // check PLLS mux has selected FLL
63  {
64  return 0x1; // return error code
65  }
66 
67  // check external frequency is less than the maximum frequency
68  if (crystal_val > 50000000) {return 0x21;}
69 
70  // check crystal frequency is within spec. if crystal osc is being used as PLL ref
71  if (erefs_val)
72  {
73  if ((crystal_val < 3000000) || (crystal_val > 32000000)) {return 0x22;} // return 1 if one of the available crystal options is not available
74  }
75 
76  // make sure HGO will never be greater than 1. Could return an error instead if desired.
77  if (hgo_val > 0)
78  {
79  hgo_val = 1; // force hgo_val to 1 if > 0
80  }
81 
82  // Check PLL divider settings are within spec.
83  if ((prdiv_val < 1) || (prdiv_val > 25)) {return 0x41;}
84  if ((vdiv_val < 24) || (vdiv_val > 50)) {return 0x42;}
85 
86  // Check PLL reference clock frequency is within spec.
87  ref_freq = crystal_val / prdiv_val;
88  if ((ref_freq < 2000000) || (ref_freq > 4000000)) {return 0x43;}
89 
90  // Check PLL output frequency is within spec.
91  pll_freq = (crystal_val / prdiv_val) * vdiv_val;
92  if ((pll_freq < 48000000) || (pll_freq > 100000000)) {return 0x45;}
93 
94  // configure the MCG_C2 register
95  // the RANGE value is determined by the external frequency. Since the RANGE parameter affects the FRDIV divide value
96  // it still needs to be set correctly even if the oscillator is not being used
97 
98  temp_reg = MCG_C2;
99  temp_reg &= ~(MCG_C2_RANGE0_MASK | MCG_C2_HGO0_MASK | MCG_C2_EREFS0_MASK); // clear fields before writing new values
100 
101  if (crystal_val <= 8000000)
102  {
103  temp_reg |= (MCG_C2_RANGE0(1) | (hgo_val << MCG_C2_HGO0_SHIFT) | (erefs_val << MCG_C2_EREFS0_SHIFT));
104  }
105  else
106  {
107  temp_reg |= (MCG_C2_RANGE0(2) | (hgo_val << MCG_C2_HGO0_SHIFT) | (erefs_val << MCG_C2_EREFS0_SHIFT));
108  }
109  MCG_C2 = temp_reg;
110 
111  // determine FRDIV based on reference clock frequency
112  // since the external frequency has already been checked only the maximum frequency for each FRDIV value needs to be compared here.
113  if (crystal_val <= 1250000) {frdiv_val = 0;}
114  else if (crystal_val <= 2500000) {frdiv_val = 1;}
115  else if (crystal_val <= 5000000) {frdiv_val = 2;}
116  else if (crystal_val <= 10000000) {frdiv_val = 3;}
117  else if (crystal_val <= 20000000) {frdiv_val = 4;}
118  else {frdiv_val = 5;}
119 
120  // Select external oscillator and Reference Divider and clear IREFS to start ext osc
121  // If IRCLK is required it must be enabled outside of this driver, existing state will be maintained
122  // CLKS=2, FRDIV=frdiv_val, IREFS=0, IRCLKEN=0, IREFSTEN=0
123  temp_reg = MCG_C1;
124  temp_reg &= ~(MCG_C1_CLKS_MASK | MCG_C1_FRDIV_MASK | MCG_C1_IREFS_MASK); // Clear values in these fields
125  temp_reg |= (MCG_C1_CLKS(2) | MCG_C1_FRDIV(frdiv_val)); // Set the required CLKS and FRDIV values
126  MCG_C1 = temp_reg;
127 
128  // if the external oscillator is used need to wait for OSCINIT to set
129  if (erefs_val)
130  {
131  for (i = 0 ; i < 20000 ; i++)
132  {
133  if (MCG_S & MCG_S_OSCINIT0_MASK) break; // jump out early if OSCINIT sets before loop finishes
134  }
135  if (!(MCG_S & MCG_S_OSCINIT0_MASK)) return 0x23; // check bit is really set and return with error if not set
136  }
137 
138  // wait for Reference clock Status bit to clear
139  for (i = 0 ; i < 2000 ; i++)
140  {
141  if (!(MCG_S & MCG_S_IREFST_MASK)) break; // jump out early if IREFST clears before loop finishes
142  }
143  if (MCG_S & MCG_S_IREFST_MASK) return 0x11; // check bit is really clear and return with error if not set
144 
145  // Wait for clock status bits to show clock source is ext ref clk
146  for (i = 0 ; i < 2000 ; i++)
147  {
148  if (((MCG_S & MCG_S_CLKST_MASK) >> MCG_S_CLKST_SHIFT) == 0x2) break; // jump out early if CLKST shows EXT CLK slected before loop finishes
149  }
150  if (((MCG_S & MCG_S_CLKST_MASK) >> MCG_S_CLKST_SHIFT) != 0x2) return 0x1A; // check EXT CLK is really selected and return with error if not
151 
152  // Now in FBE
153  // It is recommended that the clock monitor is enabled when using an external clock as the clock source/reference.
154  // It is enabled here but can be removed if this is not required.
155  MCG_C6 |= MCG_C6_CME0_MASK;
156 
157  // Configure PLL
158  // Configure MCG_C5
159  // If the PLL is to run in STOP mode then the PLLSTEN bit needs to be OR'ed in here or in user code.
160  temp_reg = MCG_C5;
161  temp_reg &= ~MCG_C5_PRDIV0_MASK;
162  temp_reg |= MCG_C5_PRDIV0(prdiv_val - 1); //set PLL ref divider
163  MCG_C5 = temp_reg;
164 
165  // Configure MCG_C6
166  // The PLLS bit is set to enable the PLL, MCGOUT still sourced from ext ref clk
167  // The loss of lock interrupt can be enabled by seperately OR'ing in the LOLIE bit in MCG_C6
168  temp_reg = MCG_C6; // store present C6 value
169  temp_reg &= ~MCG_C6_VDIV0_MASK; // clear VDIV settings
170  temp_reg |= MCG_C6_PLLS_MASK | MCG_C6_VDIV0(vdiv_val - 24); // write new VDIV and enable PLL
171  MCG_C6 = temp_reg; // update MCG_C6
172 
173  // wait for PLLST status bit to set
174  for (i = 0 ; i < 2000 ; i++)
175  {
176  if (MCG_S & MCG_S_PLLST_MASK) break; // jump out early if PLLST sets before loop finishes
177  }
178  if (!(MCG_S & MCG_S_PLLST_MASK)) return 0x16; // check bit is really set and return with error if not set
179 
180  // Wait for LOCK bit to set
181  for (i = 0 ; i < 4000 ; i++)
182  {
183  if (MCG_S & MCG_S_LOCK0_MASK) break; // jump out early if LOCK sets before loop finishes
184  }
185  if (!(MCG_S & MCG_S_LOCK0_MASK)) return 0x44; // check bit is really set and return with error if not set
186 
187  // Use actual PLL settings to calculate PLL frequency
188  prdiv = ((MCG_C5 & MCG_C5_PRDIV0_MASK) + 1);
189  vdiv = ((MCG_C6 & MCG_C6_VDIV0_MASK) + 24);
190 
191  // now in PBE
192 
193  MCG_C1 &= ~MCG_C1_CLKS_MASK; // clear CLKS to switch CLKS mux to select PLL as MCG_OUT
194 
195  // Wait for clock status bits to update
196  for (i = 0 ; i < 2000 ; i++)
197  {
198  if (((MCG_S & MCG_S_CLKST_MASK) >> MCG_S_CLKST_SHIFT) == 0x3) break; // jump out early if CLKST = 3 before loop finishes
199  }
200  if (((MCG_S & MCG_S_CLKST_MASK) >> MCG_S_CLKST_SHIFT) != 0x3) return 0x1B; // check CLKST is set correctly and return with error if not
201 
202  // Now in PEE
203 
204  return ((crystal_val / prdiv) * vdiv); //MCGOUT equals PLL output frequency
205 } // pll_init
